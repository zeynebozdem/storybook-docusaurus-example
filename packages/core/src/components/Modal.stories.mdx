## Examples

export const ModalExample = () => {
  const [isOpen, setIsOpen] = useState(false);
  
  return (
    <>
      <button onClick={() => setIsOpen(true)}>
        Modalı Aç
      </button>
      <Modal 
        title="Örnek Modal" 
        isOpen={isOpen} 
        onClose={() => setIsOpen(false)}
      >
        <p>Bu bir modal içeriğidir. Kullanıcıya bir bilgi göstermek veya bir işlem yaptırmak için kullanılır.</p>
      </Modal>
    </>
  );
};
